class Solution:
    def longestIncreasingPath(self, matrix: List[List[int]]) -> int:
        if not matrix or not matrix[0]:
            return 0
        
        m, n = len(matrix), len(matrix[0])
        directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        
        # è¨ˆç®—å‡ºåº¦
        out_degree = [[0] * n for _ in range(m)]
        for i in range(m):
            for j in range(n):
                for di, dj in directions:
                    ni, nj = i + di, j + dj
                    if 0 <= ni < m and 0 <= nj < n and matrix[ni][nj] > matrix[i][j]:

        queue = deque()

        # BFS é€å±¤å‰é›¢
        length = 0
        while queue:
            length += 1

                    ni, nj = i + di, j + dj
                    if 0 <= ni < m and 0 <= nj < n and matrix[ni][nj] < matrix[i][j]:
                        out_degree[ni][nj] -= 1
                        if out_degree[ni][nj] == 0:
                            queue.append((ni, nj))
        
        return length

    private int[][] memo;

        if (matrix.length == 0 || matrix[0].length == 0) return 0;

            for (int j = 0; j < n; j++) {
                result = Math.max(result, dfs(matrix, i, j));
            }
        }

    }

            int ni = i + dir[0];
            int nj = j + dir[1];

                matrix[ni][nj] > matrix[i][j]) {
                maxLength = Math.max(maxLength, 1 + dfs(matrix, ni, nj));
            }
        }
        
        memo[i][j] = maxLength;
        return maxLength;
    }
