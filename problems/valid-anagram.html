242. Valid Anagram
Easy

題目描述

給定兩個字串 s 和 t，編寫一個函數來判斷 t 是否是 s 的字母異位詞（anagram）。

字母異位詞是指兩個字串包含相同的字符，但排列順序可能不同。

範例 1:
輸入: s = "anagram", t = "nagaram"
輸出: true
解釋: 兩個字串包含相同的字符，只是順序不同
範例 2:
輸入: s = "rat", t = "car"
輸出: false
解釋: 字符不同，'r','a','t' vs 'c','a','r'
限制條件:
1 <= s.length, t.length <= 5 * 10^4
s 和 t 僅包含小寫英文字母
進階:

如果輸入字串包含 Unicode 字符，你該如何調整你的解法？

解題思路

方法一：排序比較

將兩個字串排序後比較是否相同。

核心思想：

如果兩個字串是 anagram，排序後應該完全相同。

步驟：

  1. 如果兩個字串長度不同，直接返回 false
  2. 將兩個字串都排序
  3. 比較排序後的字串是否相同

視覺化範例：s = "anagram", t = "nagaram"

s = "anagram"  → 排序後 → "aaagmnr"
t = "nagaram"  → 排序後 → "aaagmnr"

比較: "aaagmnr" == "aaagmnr" → true
方法二：哈希表計數（推薦）⭐

使用哈希表統計每個字符出現的次數，比較兩個字串的字符頻率。

核心思想：

如果兩個字串是 anagram，每個字符出現的次數應該完全相同。

步驟：

  1. 如果兩個字串長度不同，直接返回 false
  2. 創建哈希表統計 s 中每個字符的次數
  3. 遍歷 t，減少對應字符的計數
  4. 如果所有計數都為 0，返回 true

視覺化範例：s = "anagram", t = "nagaram"

統計 s = "anagram":
{'a': 3, 'n': 1, 'g': 1, 'r': 1, 'm': 1}

遍歷 t = "nagaram":
'n' → count['n'] = 0
'a' → count['a'] = 2
'g' → count['g'] = 0
'a' → count['a'] = 1
'r' → count['r'] = 0
'a' → count['a'] = 0
'm' → count['m'] = 0

所有計數都為 0 → true
方法三：陣列計數（最優）

利用題目限制「僅包含小寫字母」，使用長度 26 的陣列代替哈希表。

優點：

  • 空間固定 O(26) = O(1)
  • 比哈希表更快（直接陣列存取）
  • 不需要處理哈希衝突

程式碼實現

class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        """
        方法一：排序比較
        時間複雜度: O(n log n)
        空間複雜度: O(n)
        """
        return sorted(s) == sorted(t)
    
    def isAnagram_hashmap(self, s: str, t: str) -> bool:
        """
        方法二：哈希表計數（推薦）
        時間複雜度: O(n)
        空間複雜度: O(1) - 最多 26 個小寫字母
        """
        # 長度不同直接返回 false
        if len(s) != len(t):
            return False
        
        # 統計字符頻率
        count = {}
        
        # 計算 s 中每個字符的出現次數
        for char in s:
            count[char] = count.get(char, 0) + 1
        
        # 減少 t 中對應字符的計數
        for char in t:
            if char not in count:
                return False
            count[char] -= 1
            if count[char] < 0:
                return False
        
        # 檢查所有計數是否都為 0
        return all(c == 0 for c in count.values())
    
    def isAnagram_counter(self, s: str, t: str) -> bool:
        """
        方法二：使用 Counter（Python 專用）
        時間複雜度: O(n)
        空間複雜度: O(1)
        """
        from collections import Counter
        return Counter(s) == Counter(t)
    
    def isAnagram_array(self, s: str, t: str) -> bool:
        """
        方法三：陣列計數（最優）
        時間複雜度: O(n)
        空間複雜度: O(1)
        """
        if len(s) != len(t):
            return False
        
        # 長度 26 的陣列，對應 a-z
        count = [0] * 26
        
        # 統計字符差異
        for i in range(len(s)):
            count[ord(s[i]) - ord('a')] += 1
            count[ord(t[i]) - ord('a')] -= 1
        
        # 檢查是否所有計數都為 0
        return all(c == 0 for c in count)
#include <string>
#include <algorithm>
#include <unordered_map>
#include <vector>

class Solution {
public:
    /**
     * 方法一：排序比較
     * 時間複雜度: O(n log n)
     * 空間複雜度: O(n)
     */
    bool isAnagram(std::string s, std::string t) {
        std::sort(s.begin(), s.end());
        std::sort(t.begin(), t.end());
        return s == t;
    }
    
    /**
     * 方法二：哈希表計數
     * 時間複雜度: O(n)
     * 空間複雜度: O(1)
     */
    bool isAnagram_hashmap(std::string s, std::string t) {
        if (s.length() != t.length()) {
            return false;
        }
        
        std::unordered_map count;
        
        // 統計 s 中字符頻率
        for (char c : s) {
            count[c]++;
        }
        
        // 減少 t 中對應字符的計數
        for (char c : t) {
            if (count.find(c) == count.end() || count[c] == 0) {
                return false;
            }
            count[c]--;
        }
        
        return true;
    }
    
    /**
     * 方法三：陣列計數（最優）
     * 時間複雜度: O(n)
     * 空間複雜度: O(1)
     */
    bool isAnagram_array(std::string s, std::string t) {
        if (s.length() != t.length()) {
            return false;
        }
        
        std::vector count(26, 0);
        
        for (int i = 0; i < s.length(); i++) {
            count[s[i] - 'a']++;
            count[t[i] - 'a']--;
        }
        
        for (int c : count) {
            if (c != 0) return false;
        }
        
        return true;
    }
};
import java.util.Arrays;
import java.util.HashMap;
import java.util.Map;

class Solution {
    /**
     * 方法一：排序比較
     * 時間複雜度: O(n log n)
     * 空間複雜度: O(n)
     */
    public boolean isAnagram(String s, String t) {
        char[] sArray = s.toCharArray();
        char[] tArray = t.toCharArray();
        Arrays.sort(sArray);
        Arrays.sort(tArray);
        return Arrays.equals(sArray, tArray);
    }
    
    /**
     * 方法二：哈希表計數
     * 時間複雜度: O(n)
     * 空間複雜度: O(1)
     */
    public boolean isAnagram_hashmap(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        
        Map count = new HashMap<>();
        
        // 統計 s 中字符頻率
        for (char c : s.toCharArray()) {
            count.put(c, count.getOrDefault(c, 0) + 1);
        }
        
        // 減少 t 中對應字符的計數
        for (char c : t.toCharArray()) {
            if (!count.containsKey(c) || count.get(c) == 0) {
                return false;
            }
            count.put(c, count.get(c) - 1);
        }
        
        return true;
    }
    
    /**
     * 方法三：陣列計數（最優）
     * 時間複雜度: O(n)
     * 空間複雜度: O(1)
     */
    public boolean isAnagram_array(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        
        int[] count = new int[26];
        
        for (int i = 0; i < s.length(); i++) {
            count[s.charAt(i) - 'a']++;
            count[t.charAt(i) - 'a']--;
        }
        
        for (int c : count) {
            if (c != 0) return false;
        }
        
        return true;
    }
}

複雜度分析

方法一：排序比較
  • 時間複雜度：O(n log n) - 排序時間
  • 空間複雜度：O(n) - 排序需要額外空間
方法二：哈希表計數
  • 時間複雜度：O(n) - 遍歷兩次字串
  • 空間複雜度：O(1) - 最多 26 個小寫字母
方法三：陣列計數（最優）
  • 時間複雜度：O(n) - 一次遍歷
  • 空間複雜度：O(1) - 固定 26 個空間

重點提示

  • 最簡單：排序後比較（一行代碼）
  • 最優解：陣列計數 O(n) 時間 + O(1) 空間
  • 長度檢查：不同長度直接返回 false，提早剪枝
  • 字符範圍：
    • 小寫字母 → 陣列 [26]
    • 大小寫字母 → 陣列 [52]
    • Unicode → 使用哈希表
  • Python 技巧：Counter 類最簡潔
  • 面試技巧：
    • 先說排序解法（簡單）
    • 再優化到哈希表 O(n)
    • 最後提到陣列計數（空間最優）
