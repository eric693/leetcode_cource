416. Partition Equal Subset Sum
Medium

題目描述

給定一個只包含正整數的非空陣列 nums，判斷是否可以將這個陣列分割成兩個子集，使得兩個子集的元素和相等。

範例 1:
輸入: nums = [1,5,11,5]
輸出: true
解釋: 陣列可以分割成 [1, 5, 5] 和 [11]。
範例 2:
輸入: nums = [1,2,3,5]
輸出: false
解釋: 陣列不能分割成兩個和相等的子集。
限制條件:
1 <= nums.length <= 200
1 <= nums[i] <= 100

解題思路

核心觀察

這題可以轉化為經典的0/1 背包問題：

  • 如果陣列總和為奇數 → 不可能平分 → 返回 false
  • 如果陣列總和為偶數 → 問題變成：能否從陣列中選出元素，使其和等於 total/2

轉化後問題：能否在陣列中找到子集，使其和等於 target = sum(nums) / 2

方法一：二維動態規劃（推薦理解）⭐

使用二維 DP，dp[i][j] 表示使用前 i 個數字，能否湊出和為 j。

狀態定義：

  • dp[i][j]：使用前 i 個數字，能否湊出和為 j
  • 初始狀態：dp[i][0] = true（和為 0，不選任何數字）
  • 轉移方程：
    • 不選第 i 個數：dp[i][j] = dp[i-1][j]
    • 選第 i 個數：dp[i][j] = dp[i-1][j-nums[i]]（如果 j >= nums[i]）
    • 兩者取 OR：dp[i][j] = dp[i-1][j] || dp[i-1][j-nums[i]]

視覺化範例：nums = [1,5,11,5], target = 11

陣列: [1, 5, 11, 5]
總和: 22，目標: 11

DP 表格（行=前i個數，列=目標和j）:

dp[4][11] = T，可以湊出 11！
方案：[1, 5, 5] 或 [11]
方法二：一維動態規劃（空間優化）⭐⭐

觀察到 dp[i] 只依賴 dp[i-1]，可以用一維陣列優化空間。

關鍵技巧：

  • 從右往左更新（避免覆蓋還需要使用的值）
  • dp[j] = dp[j] || dp[j - num]

一維 DP 過程：nums = [1,5,11,5], target = 11

初始: dp = [T, F, F, F, F, F, F, F, F, F, F, F]

處理 1:
     dp = [T, T, F, F, F, F, F, F, F, F, F, F]
           ↑  ↑ (j=1時，dp[1] = dp[1] || dp[0])

處理 5:
     dp = [T, T, F, F, F, T, T, F, F, F, F, F]
                       ↑  ↑ (可以湊出5和6)

處理 11:
     dp = [T, T, F, F, F, T, T, F, F, F, F, T]
                                         ↑ (可以湊出11)

處理 5:
     dp = [T, T, F, F, F, T, T, F, F, F, T, T]
                                   ↑ (可以湊出10)

結果: dp[11] = T
提前終止優化
  • 總和為奇數：直接返回 false
  • 最大元素 > target：不可能分割
  • 找到答案：dp[target] = true 時可以立即返回

程式碼實現

class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        """
        方法：一維動態規劃（推薦）
        時間複雜度: O(n * target)
        空間複雜度: O(target)
        """
        total = sum(nums)
        
        # 總和為奇數，不可能平分
        if total % 2 != 0:
            return False
        
        target = total // 2
        
        # dp[j] 表示能否湊出和為 j
        dp = [False] * (target + 1)
        dp[0] = True  # 和為 0
        
        # 遍歷每個數字
        for num in nums:
            # 從右往左更新（避免重複使用同一個數）
            for j in range(target, num - 1, -1):
                dp[j] = dp[j] or dp[j - num]
        
        return dp[target]
    
    def canPartition_2d(self, nums: List[int]) -> bool:
        """
        方法：二維動態規劃（更容易理解）
        時間複雜度: O(n * target)
        空間複雜度: O(n * target)
        """
        total = sum(nums)
        
        if total % 2 != 0:
            return False
        
        target = total // 2
        n = len(nums)
        
        # dp[i][j] 表示使用前 i 個數能否湊出和 j
        dp = [[False] * (target + 1) for _ in range(n + 1)]
        
        # 初始化：和為 0 總是可以（不選任何數）
        for i in range(n + 1):
            dp[i][0] = True
        
        # 填充 DP 表
        for i in range(1, n + 1):
            for j in range(1, target + 1):
                # 不選第 i 個數
                dp[i][j] = dp[i-1][j]
                
                # 選第 i 個數（如果可以）
                if j >= nums[i-1]:
                    dp[i][j] = dp[i][j] or dp[i-1][j - nums[i-1]]
        
        return dp[n][target]
    
    def canPartition_optimized(self, nums: List[int]) -> bool:
        """
        優化版：加入提前終止
        """
        total = sum(nums)
        
        if total % 2 != 0:
            return False
        
        target = total // 2
        max_num = max(nums)
        
        # 如果最大數大於目標，無法分割
        if max_num > target:
            return False
        
        # 如果最大數等於目標，直接成功
        if max_num == target:
            return True
        
        dp = [False] * (target + 1)
        dp[0] = True
        
        for num in nums:
            for j in range(target, num - 1, -1):
                dp[j] = dp[j] or dp[j - num]
                # 提前終止
                if dp[target]:
                    return True
        
        return dp[target]
    
    def canPartition_set(self, nums: List[int]) -> bool:
        """
        使用 Set 的簡潔寫法
        """
        total = sum(nums)
        
        if total % 2:
            return False
        
        target = total // 2
        possible = {0}  # 可能湊出的和
        
        for num in nums:
            # 更新所有可能的和
            possible = possible | {p + num for p in possible}
            
            # 提前終止
            if target in possible:
                return True
        
        return target in possible

#include <numeric>
using namespace std;

class Solution {
public:
    /**
     * 方法：一維動態規劃
     * 時間複雜度: O(n * target)
     * 空間複雜度: O(target)
     */
    bool canPartition(vector& nums) {
        int total = accumulate(nums.begin(), nums.end(), 0);
        
        // 總和為奇數
        if (total % 2 != 0) {
            return false;
        }
        
        int target = total / 2;
        vector dp(target + 1, false);
        dp[0] = true;
        
        for (int num : nums) {
            // 從右往左更新
            for (int j = target; j >= num; j--) {
                dp[j] = dp[j] || dp[j - num];
            }
        }
        
        return dp[target];
    }
    
    /**
     * 優化版：加入提前終止
     */
    bool canPartition_optimized(vector& nums) {
        int total = accumulate(nums.begin(), nums.end(), 0);
        
        if (total % 2 != 0) {
            return false;
        }
        
        int target = total / 2;
        int maxNum = *max_element(nums.begin(), nums.end());
        
        if (maxNum > target) {
            return false;
        }
        
        if (maxNum == target) {
            return true;
        }
        
        vector dp(target + 1, false);
        dp[0] = true;
        
        for (int num : nums) {
            for (int j = target; j >= num; j--) {
                dp[j] = dp[j] || dp[j - num];
                if (dp[target]) {
                    return true;
                }
            }
        }
        
        return dp[target];
    }
};

        int total = 0;
        for (int num : nums) {
            total += num;
        }
        
        // 總和為奇數
        if (total % 2 != 0) {
            return false;
        }
        
        int target = total / 2;
        boolean[] dp = new boolean[target + 1];
        dp[0] = true;
        
        for (int num : nums) {
            // 從右往左更新
            for (int j = target; j >= num; j--) {
                dp[j] = dp[j] || dp[j - num];
            }
        }
        
        return dp[target];
    }
    
    /**
     * 優化版：加入提前終止
     */
    public boolean canPartition_optimized(int[] nums) {
        int total = 0;
        int maxNum = 0;
        
        for (int num : nums) {
            total += num;
            maxNum = Math.max(maxNum, num);
        }
        
        if (total % 2 != 0) {
            return false;
        }
        
        int target = total / 2;

            return false;

            return true;

        boolean[] dp = new boolean[target + 1];
        dp[0] = true;

            for (int j = target; j >= num; j--) {
                dp[j] = dp[j] || dp[j - num];
                if (dp[target]) {

                }
            }
        }
        
        return dp[target];

複雜度分析

時間複雜度: O(n × target)

n 是陣列長度，target = sum(nums) / 2。需要填充 n × target 個狀態。 最壞情況下 target 可達 10000（200個數，每個100）。

空間複雜度: O(target)

一維 DP 陣列需要 O(target) 空間。 二維 DP 需要 O(n × target) 空間。

重點提示

  • 問題轉化：分割問題 → 0/1 背包問題
  • 關鍵觀察：
    • 總和為奇數 → 必定無法平分
    • 只需要找到一個子集和 = total/2
    • 這是典型的 0/1 背包（每個數只能用一次）
  • DP 狀態設計：
    • 二維：dp[i][j] = 前 i 個數能否湊出 j
    • 一維：dp[j] = 能否湊出 j
    • 一維必須從右往左更新！
  • 為什麼從右往左：
    • 保證每個數只用一次
    • 避免用更新後的值再次更新
    • 相當於滾動陣列優化
  • 優化技巧：
    • 檢查總和奇偶性
    • 檢查最大元素
    • 找到答案立即返回
    • 排序後從大到小處理（可能更快找到答案）
  • 邊界情況：
    • 只有一個元素 → false
    • 兩個相同元素 → true
    • 所有元素相同 → 看個數奇偶
    • target = 0 → true（但題目保證正整數）
  • 常見錯誤：
    • 一維 DP 從左往右更新
    • 忘記檢查總和奇偶性
    • dp[0] 沒有初始化為 true
  • 面試技巧：
    • 先說明問題轉化：分割 → 背包
    • 解釋為什麼檢查 sum/2
    • 從二維 DP 推導到一維優化
    • 強調從右往左的重要性
