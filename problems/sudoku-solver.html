class Solution:
    def solveSudoku(self, board: List[List[str]]) -> None:
        # ä½¿ç”¨ä½æ©ç¢¼è¡¨ç¤ºå¯ç”¨æ•¸å­—ï¼ˆ1-9ï¼‰
        rows = [0] * 9
        cols = [0] * 9
        boxes = [0] * 9
        
        # é è™•ç†
        for i in range(9):
            for j in range(9):
                if board[i][j] != '.':
                    digit = int(board[i][j]) - 1
                    box_idx = (i // 3) * 3 + j // 3
                    # è¨­ç½®å°æ‡‰ä½ç‚º 1
                    rows[i] |= (1 << digit)
                    cols[j] |= (1 << digit)
                    boxes[box_idx] |= (1 << digit)
        
        def backtrack(pos):
            if pos == 81:
                return True
            
            row, col = pos // 9, pos % 9
            
            if board[row][col] != '.':
                return backtrack(pos + 1)
            
            box_idx = (row // 3) * 3 + col // 3
            # è¨ˆç®—å¯ç”¨æ•¸å­—ï¼ˆæœªè¢«ä½¿ç”¨çš„ä½ç‚º 1ï¼‰
            mask = ~(rows[row] | cols[col] | boxes[box_idx]) & 0x1ff
            
            while mask:
                # ç²å–æœ€ä½ä½çš„ 1
                digit = (mask & -mask).bit_length() - 1
                num = str(digit + 1)
                
                # åšé¸æ“‡
                board[row][col] = num
                rows[row] |= (1 << digit)
                cols[col] |= (1 << digit)
                boxes[box_idx] |= (1 << digit)
                
                if backtrack(pos + 1):
                    return True
                
                # æ’¤éŠ·
                board[row][col] = '.'
                rows[row] ^= (1 << digit)
                cols[col] ^= (1 << digit)
                boxes[box_idx] ^= (1 << digit)
                
                # æ¸…é™¤æœ€ä½ä½çš„ 1
                mask &= (mask - 1)
            
            return False
        
        backtrack(0)

class Solution {
public:
    void solveSudoku(vector>& board) {
        vector> rows(9);
        vector> cols(9);
        vector> boxes(9);
        
        // é è™•ç†
        for (int i = 0; i < 9; i++) {
            for (int j = 0; j < 9; j++) {
                if (board[i][j] != '.') {
                    char num = board[i][j];
                    rows[i].insert(num);
                    cols[j].insert(num);
                    int boxIdx = (i / 3) * 3 + j / 3;
                    boxes[boxIdx].insert(num);
                }
            }
        }
        
        backtrack(board, 0, rows, cols, boxes);
    }
    
private:
    bool backtrack(vector>& board, int pos,
                   vector>& rows,
                   vector>& cols,
                   vector>& boxes) {
        if (pos == 81) return true;
        
        int row = pos / 9;
        int col = pos % 9;
        
        if (board[row][col] != '.') {
            return backtrack(board, pos + 1, rows, cols, boxes);
        }
        
        int boxIdx = (row / 3) * 3 + col / 3;
        
        for (char num = '1'; num <= '9'; num++) {
            if (rows[row].count(num) || 
                cols[col].count(num) || 
                boxes[boxIdx].count(num)) {
                continue;
            }
            
            board[row][col] = num;
            rows[row].insert(num);
            cols[col].insert(num);
            boxes[boxIdx].insert(num);
            
            if (backtrack(board, pos + 1, rows, cols, boxes)) {
                return true;
            }
            
            board[row][col] = '.';
            rows[row].erase(num);
            cols[col].erase(num);
            boxes[boxIdx].erase(num);
        }
        
        return false;
    }
};

class Solution {
    public void solveSudoku(char[][] board) {
        Set[] rows = new HashSet[9];
        Set[] cols = new HashSet[9];
        Set[] boxes = new HashSet[9];
        
        for (int i = 0; i < 9; i++) {
            rows[i] = new HashSet<>();
            cols[i] = new HashSet<>();
            boxes[i] = new HashSet<>();
        }
        
        // é è™•ç†
        for (int i = 0; i < 9; i++) {
            for (int j = 0; j < 9; j++) {
                if (board[i][j] != '.') {
                    char num = board[i][j];
                    rows[i].add(num);
                    cols[j].add(num);
                    int boxIdx = (i / 3) * 3 + j / 3;
                    boxes[boxIdx].add(num);
                }
            }
        }
        
        backtrack(board, 0, rows, cols, boxes);
    }
    
    private boolean backtrack(char[][] board, int pos,
                             Set[] rows,
                             Set[] cols,
                             Set[] boxes) {
        if (pos == 81) return true;
        
        int row = pos / 9;
        int col = pos % 9;
        
        if (board[row][col] != '.') {
            return backtrack(board, pos + 1, rows, cols, boxes);
        }
        
        int boxIdx = (row / 3) * 3 + col / 3;
        
        for (char num = '1'; num <= '9'; num++) {
            if (rows[row].contains(num) || 
                cols[col].contains(num) || 
                boxes[boxIdx].contains(num)) {
                continue;
            }
            
            board[row][col] = num;
            rows[row].add(num);
            cols[col].add(num);
            boxes[boxIdx].add(num);
            
            if (backtrack(board, pos + 1, rows, cols, boxes)) {
                return true;
            }
            
            board[row][col] = '.';
            rows[row].remove(num);
            cols[col].remove(num);
            boxes[boxIdx].remove(num);
        }
        
        return false;
    }
}
