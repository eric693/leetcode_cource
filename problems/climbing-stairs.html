70. Climbing Stairs
Easy

題目描述

假設你正在爬樓梯。需要 n 階你才能到達樓頂。

每次你可以爬 1 或 2 個台階。你有多少種不同的方法可以爬到樓頂呢？

範例 1:
輸入: n = 2
輸出: 2
解釋: 有兩種方法可以爬到樓頂。
1. 1 階 + 1 階
2. 2 階
範例 2:
輸入: n = 3
輸出: 3
解釋: 有三種方法可以爬到樓頂。
1. 1 階 + 1 階 + 1 階
2. 1 階 + 2 階
3. 2 階 + 1 階
範例 3:
輸入: n = 4
輸出: 5
解釋: 有五種方法可以爬到樓頂。
1. 1+1+1+1
2. 1+1+2
3. 1+2+1
4. 2+1+1
5. 2+2
限制條件:
1 <= n <= 45

解題思路

關鍵觀察：這是斐波那契數列！

要到達第 n 階，可以從第 (n-1) 階走 1 步，或從第 (n-2) 階走 2 步。

狀態轉移方程：

dp[n] = dp[n-1] + dp[n-2]

視覺化推導（n = 5）：

n = 1: 1 種方法 (1)
n = 2: 2 種方法 (1+1, 2)
n = 3: 3 種方法 = dp[2] + dp[1] = 2 + 1
n = 4: 5 種方法 = dp[3] + dp[2] = 3 + 2
n = 5: 8 種方法 = dp[4] + dp[3] = 5 + 3

數列: 1, 2, 3, 5, 8, 13, 21...（斐波那契數列）
方法一：動態規劃（DP 陣列）

使用一個陣列儲存每一階的方法數，從底向上計算。

步驟：

  1. 創建 dp 陣列，長度為 n+1
  2. 初始化基礎情況：
    • dp[1] = 1（到第 1 階只有 1 種方法）
    • dp[2] = 2（到第 2 階有 2 種方法）
  3. 從 i = 3 到 n，計算 dp[i] = dp[i-1] + dp[i-2]
  4. 返回 dp[n]
方法二：動態規劃（空間優化）

觀察到我們只需要前兩個狀態，可以用兩個變數代替整個陣列。

優化思路：

由於 dp[i] 只依賴於 dp[i-1] 和 dp[i-2]，我們不需要保存整個陣列，只需要兩個變數即可。

步驟：

  1. 初始化 prev1 = 2（第 2 階），prev2 = 1（第 1 階）
  2. 從 i = 3 到 n：
    • current = prev1 + prev2
    • 更新：prev2 = prev1, prev1 = current
  3. 返回 prev1
方法三：遞迴 + 記憶化（Memoization）

使用遞迴思路，並用字典或陣列快取已計算的結果，避免重複計算。

步驟：

  1. 定義遞迴函數 climb(n)：
    • 基礎情況：n ≤ 2 時返回 n
    • 如果 n 已在快取中，直接返回
    • 否則計算：climb(n-1) + climb(n-2)
    • 將結果存入快取並返回

程式碼實現

class Solution:
    def climbStairs(self, n: int) -> int:
        """
        方法一：動態規劃（DP 陣列）
        時間複雜度: O(n)
        空間複雜度: O(n)
        """
        if n <= 2:
            return n
        
        # dp[i] 表示到達第 i 階的方法數
        dp = [0] * (n + 1)
        dp[1] = 1
        dp[2] = 2
        
        for i in range(3, n + 1):
            dp[i] = dp[i-1] + dp[i-2]
        
        return dp[n]
    
    def climbStairs_optimized(self, n: int) -> int:
        """
        方法二：動態規劃（空間優化）
        時間複雜度: O(n)
        空間複雜度: O(1)
        """
        if n <= 2:
            return n
        
        # 只需要保存前兩個狀態
        prev2 = 1  # dp[1]
        prev1 = 2  # dp[2]
        
        for i in range(3, n + 1):
            current = prev1 + prev2
            prev2 = prev1
            prev1 = current
        
        return prev1
    
    def climbStairs_memo(self, n: int) -> int:
        """
        方法三：遞迴 + 記憶化
        時間複雜度: O(n)
        空間複雜度: O(n)
        """
        memo = {}
        
        def climb(n):
            if n <= 2:
                return n
            
            if n in memo:
                return memo[n]
            
            memo[n] = climb(n-1) + climb(n-2)
            return memo[n]
        
        return climb(n)
    
    def climbStairs_clean(self, n: int) -> int:
        """
        方法四：最簡潔寫法（推薦）
        時間複雜度: O(n)
        空間複雜度: O(1)
        """
        if n <= 2:
            return n
        
        a, b = 1, 2
        for _ in range(3, n + 1):
            a, b = b, a + b
        
        return b
#include <vector>
#include <unordered_map>

class Solution {
public:
    /**
     * 方法一：動態規劃（DP 陣列）
     * 時間複雜度: O(n)
     * 空間複雜度: O(n)
     */
    int climbStairs(int n) {
        if (n <= 2) {
            return n;
        }
        
        // dp[i] 表示到達第 i 階的方法數
        std::vector dp(n + 1);
        dp[1] = 1;
        dp[2] = 2;
        
        for (int i = 3; i <= n; i++) {
            dp[i] = dp[i-1] + dp[i-2];
        }
        
        return dp[n];
    }
    
    /**
     * 方法二：動態規劃（空間優化）
     * 時間複雜度: O(n)
     * 空間複雜度: O(1)
     */
    int climbStairs_optimized(int n) {
        if (n <= 2) {
            return n;
        }
        
        // 只需要保存前兩個狀態
        int prev2 = 1;  // dp[1]
        int prev1 = 2;  // dp[2]
        
        for (int i = 3; i <= n; i++) {
            int current = prev1 + prev2;
            prev2 = prev1;
            prev1 = current;
        }
        
        return prev1;
    }
    
    /**
     * 方法三：遞迴 + 記憶化
     * 時間複雜度: O(n)
     * 空間複雜度: O(n)
     */
    int climbStairs_memo(int n) {
        std::unordered_map memo;
        return climb(n, memo);
    }
    
private:
    int climb(int n, std::unordered_map& memo) {
        if (n <= 2) {
            return n;
        }
        
        if (memo.find(n) != memo.end()) {
            return memo[n];
        }
        
        memo[n] = climb(n-1, memo) + climb(n-2, memo);
        return memo[n];
    }

public:
    /**
     * 方法四：最簡潔寫法（推薦）
     * 時間複雜度: O(n)
     * 空間複雜度: O(1)
     */
    int climbStairs_clean(int n) {
        if (n <= 2) {
            return n;
        }
        
        int a = 1, b = 2;
        for (int i = 3; i <= n; i++) {
            int temp = a + b;
            a = b;
            b = temp;
        }
        
        return b;
    }
};
import java.util.HashMap;
import java.util.Map;

class Solution {
    /**
     * 方法一：動態規劃（DP 陣列）
     * 時間複雜度: O(n)
     * 空間複雜度: O(n)
     */
    public int climbStairs(int n) {
        if (n <= 2) {
            return n;
        }
        
        // dp[i] 表示到達第 i 階的方法數
        int[] dp = new int[n + 1];
        dp[1] = 1;
        dp[2] = 2;
        
        for (int i = 3; i <= n; i++) {
            dp[i] = dp[i-1] + dp[i-2];
        }
        
        return dp[n];
    }
    
    /**
     * 方法二：動態規劃（空間優化）
     * 時間複雜度: O(n)
     * 空間複雜度: O(1)
     */
    public int climbStairs_optimized(int n) {
        if (n <= 2) {
            return n;
        }
        
        // 只需要保存前兩個狀態
        int prev2 = 1;  // dp[1]
        int prev1 = 2;  // dp[2]
        
        for (int i = 3; i <= n; i++) {
            int current = prev1 + prev2;
            prev2 = prev1;
            prev1 = current;
        }
        
        return prev1;
    }
    
    /**
     * 方法三：遞迴 + 記憶化
     * 時間複雜度: O(n)
     * 空間複雜度: O(n)
     */
    public int climbStairs_memo(int n) {
        Map memo = new HashMap<>();
        return climb(n, memo);
    }
    
    private int climb(int n, Map memo) {
        if (n <= 2) {
            return n;
        }
        
        if (memo.containsKey(n)) {
            return memo.get(n);
        }
        
        int result = climb(n-1, memo) + climb(n-2, memo);
        memo.put(n, result);
        return result;
    }
    
    /**
     * 方法四：最簡潔寫法（推薦）
     * 時間複雜度: O(n)
     * 空間複雜度: O(1)
     */
    public int climbStairs_clean(int n) {
        if (n <= 2) {
            return n;
        }
        
        int a = 1, b = 2;
        for (int i = 3; i <= n; i++) {
            int temp = a + b;
            a = b;
            b = temp;
        }
        
        return b;
    }
}

複雜度分析

時間複雜度: O(n)

所有方法都需要計算 n 個狀態，每個狀態的計算都是 O(1)，所以總時間複雜度為 O(n)。

空間複雜度:
  • 方法一（DP 陣列）：O(n) - 需要長度為 n+1 的陣列
  • 方法二（空間優化）：O(1) - 只需要兩個變數
  • 方法三（記憶化）：O(n) - 需要 HashMap 存儲 n 個狀態，加上遞迴堆疊 O(n)
  • 方法四（最簡潔）：O(1) - 只需要兩個變數

重點提示

  • 動態規劃入門題：這是學習 DP 最經典的入門題目，務必掌握
  • 斐波那契本質：這題本質上就是求第 n 個斐波那契數
  • 狀態定義：dp[i] = 到達第 i 階的方法總數
  • 狀態轉移：dp[i] = dp[i-1] + dp[i-2]（可以從前一階或前兩階到達）
  • 空間優化：滾動變數技巧，是 DP 優化的常用手法
  • 避免陷阱：
    • 不要用純遞迴（會 TLE，時間複雜度 O(2^n)）
    • 記得處理 n=1 和 n=2 的基礎情況
  • 擴展思考：如果每次可以爬 1, 2, 3 階呢？狀態轉移會變成 dp[i] = dp[i-1] + dp[i-2] + dp[i-3]
