class Solution {
public:
    vector> solveNQueens(int n) {
        vector> result;
        vector board(n, string(n, '.'));
        
        unordered_set cols;
        unordered_set diag1;
        unordered_set diag2;
        
        backtrack(0, n, board, result, cols, diag1, diag2);
        return result;
    }
    
private:
    void backtrack(int row, int n, vector& board,
                   vector>& result,
                   unordered_set& cols,
                   unordered_set& diag1,
                   unordered_set& diag2) {
        if (row == n) {
            result.push_back(board);
            return;
        }
        
        for (int col = 0; col < n; col++) {
            if (cols.count(col) || 
                diag1.count(row - col) || 
                diag2.count(row + col)) {
                continue;
            }
            
            board[row][col] = 'Q';
            cols.insert(col);
            diag1.insert(row - col);
            diag2.insert(row + col);
            
            backtrack(row + 1, n, board, result, cols, diag1, diag2);
            
            board[row][col] = '.';
            cols.erase(col);
            diag1.erase(row - col);
            diag2.erase(row + col);
        }
    }
};

class Solution {
    public List> solveNQueens(int n) {
        List> result = new ArrayList<>();
        char[][] board = new char[n][n];
        for (int i = 0; i < n; i++) {
            Arrays.fill(board[i], '.');
        }
        
        Set cols = new HashSet<>();
        Set diag1 = new HashSet<>();
        Set diag2 = new HashSet<>();
        
        backtrack(0, n, board, result, cols, diag1, diag2);
        return result;
    }
    
    private void backtrack(int row, int n, char[][] board,
                          List> result,
                          Set cols,
                          Set diag1,
                          Set diag2) {
        if (row == n) {
            result.add(construct(board));
            return;
        }
        
        for (int col = 0; col < n; col++) {
            if (cols.contains(col) || 
                diag1.contains(row - col) || 
                diag2.contains(row + col)) {
                continue;
            }
            
            board[row][col] = 'Q';
            cols.add(col);
            diag1.add(row - col);
            diag2.add(row + col);
            
            backtrack(row + 1, n, board, result, cols, diag1, diag2);
            
            board[row][col] = '.';
            cols.remove(col);
            diag1.remove(row - col);
            diag2.remove(row + col);
        }
    }
    
    private List construct(char[][] board) {
        List result = new ArrayList<>();
        for (char[] row : board) {
            result.add(new String(row));
        }
        return result;
    }
}
