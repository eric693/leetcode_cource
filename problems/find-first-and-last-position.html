34. Find First and Last Position of Element in Sorted Array
Medium

題目描述

給定一個按照升序排列的整數陣列 nums，和一個目標值 target。找出給定目標值在陣列中的開始位置和結束位置。

如果陣列中不存在目標值 target，返回 [-1, -1]。

進階：你可以設計並實現時間複雜度為 O(log n) 的演算法解決此問題嗎？

範例 1:
輸入: nums = [5,7,7,8,8,10], target = 8
輸出: [3,4]
範例 2:
輸入: nums = [5,7,7,8,8,10], target = 6
輸出: [-1,-1]
範例 3:
輸入: nums = [], target = 0
輸出: [-1,-1]
限制條件:
0 <= nums.length <= 10^5
-10^9 <= nums[i] <= 10^9
nums 是一個非遞減陣列
-10^9 <= target <= 10^9

解題思路

方法：兩次二分搜尋（推薦）⭐

使用兩次二分搜尋，分別找到目標值的左邊界和右邊界。

核心思想：

第一次二分搜尋找最左邊的 target，第二次找最右邊的 target。

找左邊界的邏輯：

  • 當 nums[mid] == target 時，不要返回，而是繼續向左搜尋
  • 這樣可以找到最左邊的 target

找右邊界的邏輯：

  • 當 nums[mid] == target 時，不要返回，而是繼續向右搜尋
  • 這樣可以找到最右邊的 target

視覺化範例：nums = [5,7,7,8,8,10], target = 8

陣列: [5, 7, 7, 8, 8, 10]
索引:  0  1  2  3  4  5

====== 找左邊界 ======
初始: left=0, right=5

步驟 1: mid=2, nums[2]=7 < 8
       向右搜尋 → left=3

步驟 2: mid=4, nums[4]=8 == 8
       找到了，但繼續向左搜尋 → right=3

步驟 3: mid=3, nums[3]=8 == 8
       繼續向左搜尋 → right=2

步驟 4: left > right，停止
       左邊界 = 3

====== 找右邊界 ======
初始: left=0, right=5

步驟 1: mid=2, nums[2]=7 < 8
       向右搜尋 → left=3

步驟 2: mid=4, nums[4]=8 == 8
       找到了，但繼續向右搜尋 → left=5

步驟 3: mid=5, nums[5]=10 > 8
       向左搜尋 → right=4

步驟 4: left > right，停止
       右邊界 = 4

結果: [3, 4]
二分搜尋的關鍵修改

標準二分搜尋：

if nums[mid] == target:
    return mid  # 找到就返回

找左邊界：

if nums[mid] >= target:
    right = mid - 1  # 繼續向左

找右邊界：

if nums[mid] <= target:
    left = mid + 1  # 繼續向右

程式碼實現

class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        """
        方法：兩次二分搜尋
        時間複雜度: O(log n)
        空間複雜度: O(1)
        """
        def findLeft(nums, target):
            """找左邊界"""
            left, right = 0, len(nums) - 1
            result = -1
            
            while left <= right:
                mid = left + (right - left) // 2
                
                if nums[mid] == target:
                    result = mid
                    right = mid - 1  # 繼續向左搜尋
                elif nums[mid] < target:
                    left = mid + 1
                else:
                    right = mid - 1
            
            return result
        
        def findRight(nums, target):
            """找右邊界"""
            left, right = 0, len(nums) - 1
            result = -1
            
            while left <= right:
                mid = left + (right - left) // 2
                
                if nums[mid] == target:
                    result = mid
                    left = mid + 1  # 繼續向右搜尋
                elif nums[mid] < target:
                    left = mid + 1
                else:
                    right = mid - 1
            
            return result
        
        # 空陣列直接返回
        if not nums:
            return [-1, -1]
        
        left_bound = findLeft(nums, target)
        
        # 如果找不到左邊界，說明 target 不存在
        if left_bound == -1:
            return [-1, -1]
        
        right_bound = findRight(nums, target)
        
        return [left_bound, right_bound]
#include <vector>

class Solution {
public:
    /**
     * 方法：兩次二分搜尋
     * 時間複雜度: O(log n)
     * 空間複雜度: O(1)
     */
    std::vector searchRange(std::vector& nums, int target) {
        if (nums.empty()) {
            return {-1, -1};
        }
        
        int leftBound = findLeft(nums, target);
        
        if (leftBound == -1) {
            return {-1, -1};
        }
        
        int rightBound = findRight(nums, target);
        
        return {leftBound, rightBound};
    }
    
private:
    int findLeft(const std::vector& nums, int target) {
        int left = 0, right = nums.size() - 1;
        int result = -1;
        
        while (left <= right) {
            int mid = left + (right - left) / 2;
            
            if (nums[mid] == target) {
                result = mid;
                right = mid - 1;  // 繼續向左搜尋
            } else if (nums[mid] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        
        return result;
    }
    
    int findRight(const std::vector& nums, int target) {
        int left = 0, right = nums.size() - 1;
        int result = -1;
        
        while (left <= right) {
            int mid = left + (right - left) / 2;
            
            if (nums[mid] == target) {
                result = mid;
                left = mid + 1;  // 繼續向右搜尋
            } else if (nums[mid] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        
        return result;
    }
};
class Solution {
    /**
     * 方法：兩次二分搜尋
     * 時間複雜度: O(log n)
     * 空間複雜度: O(1)
     */
    public int[] searchRange(int[] nums, int target) {
        if (nums.length == 0) {
            return new int[]{-1, -1};
        }
        
        int leftBound = findLeft(nums, target);
        
        if (leftBound == -1) {
            return new int[]{-1, -1};
        }
        
        int rightBound = findRight(nums, target);
        
        return new int[]{leftBound, rightBound};
    }
    
    private int findLeft(int[] nums, int target) {
        int left = 0, right = nums.length - 1;
        int result = -1;
        
        while (left <= right) {
            int mid = left + (right - left) / 2;
            
            if (nums[mid] == target) {
                result = mid;
                right = mid - 1;  // 繼續向左搜尋
            } else if (nums[mid] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        
        return result;
    }
    
    private int findRight(int[] nums, int target) {
        int left = 0, right = nums.length - 1;
        int result = -1;
        
        while (left <= right) {
            int mid = left + (right - left) / 2;
            
            if (nums[mid] == target) {
                result = mid;
                left = mid + 1;  // 繼續向右搜尋
            } else if (nums[mid] < target) {
                left = mid + 1;
            } else {

            }
        }
        
        return result;
    }
}

複雜度分析

時間複雜度: O(log n)

執行兩次二分搜尋，每次 O(log n)，總共 O(log n)。

空間複雜度: O(1)

只使用常數個變數。

重點提示

  • 二分搜尋變形：這是學習二分搜尋邊界問題的經典題
  • 為什麼需要兩次搜尋：一次找左邊界，一次找右邊界
  • 關鍵差異：
    • 標準二分：找到就返回
    • 找左邊界：找到後繼續向左
    • 找右邊界：找到後繼續向右
  • 優化技巧：如果找不到左邊界，直接返回 [-1,-1]
  • 邊界情況：
    • 空陣列 → [-1,-1]
    • target 不存在 → [-1,-1]
    • 只有一個 target → [i,i]
    • 全是 target → [0,n-1]
  • 常見錯誤：
    • 找到就返回（應該繼續搜尋）
    • 邊界條件錯誤（left <= right）
    • 整數溢位（使用 mid = left + (right-left)//2）
  • 面試技巧：
    • 先說清楚需要兩次二分搜尋
    • 解釋為什麼找到後不能立即返回
    • 畫圖說明左右邊界的搜尋過程
