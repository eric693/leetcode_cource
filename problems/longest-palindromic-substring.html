5. Longest Palindromic Substring
Medium

題目描述

給定一個字串 s，找出 s 中最長的回文子字串。

回文串是指正著讀和反著讀都一樣的字串，例如 "aba" 或 "abba"。

範例 1:
輸入: s = "babad"
輸出: "bab"
解釋: "aba" 也是有效答案
範例 2:
輸入: s = "cbbd"
輸出: "bb"
限制條件:
1 <= s.length <= 1000
s 僅由數字和英文字母組成

解題思路

方法一：中心擴展法（推薦）⭐

從每個可能的中心點向兩邊擴展，找出最長的回文。

核心思想：

回文有兩種類型：

  • 奇數長度：中心是一個字符，如 "aba"
  • 偶數長度：中心是兩個字符，如 "abba"

步驟：

  1. 遍歷每個位置作為中心點
  2. 對每個中心：
    • 以單個字符為中心擴展（奇數長度）
    • 以兩個字符為中心擴展（偶數長度）
  3. 記錄最長的回文子字串

視覺化範例：s = "babad"

索引: 0 1 2 3 4
字串: b a b a d

中心 0 (b):
  奇數: "b" (長度 1)
  偶數: "" (不匹配)

中心 1 (a):
  奇數: "bab" (長度 3) ✓ 最長
  偶數: "" (不匹配)

中心 2 (b):
  奇數: "aba" (長度 3) ✓ 也是最長
  偶數: "" (不匹配)

中心 3 (a):
  奇數: "a" (長度 1)
  偶數: "" (不匹配)

中心 4 (d):
  奇數: "d" (長度 1)

答案: "bab" 或 "aba" (長度 3)
方法二：動態規劃

使用 dp[i][j] 表示 s[i:j+1] 是否為回文。

狀態轉移方程：

dp[i][j] = (s[i] == s[j]) AND (j - i < 3 OR dp[i+1][j-1])

解釋：

  • 如果 s[i] != s[j]，一定不是回文
  • 如果 s[i] == s[j]：
    • 長度 ≤ 3 → 一定是回文
    • 長度 > 3 → 檢查內部是否為回文
方法三：Manacher's Algorithm（進階）

最優解，O(n) 時間複雜度的專門算法。

面試中很少要求，但了解思想有助於加分。

方法對比
方法 時間 空間 推薦度
中心擴展 O(n²) O(1) ⭐⭐⭐⭐⭐
動態規劃 O(n²) O(n²) ⭐⭐⭐
Manacher O(n) O(n) ⭐⭐ (太複雜)

程式碼實現

class Solution:
    def longestPalindrome(self, s: str) -> str:
        """
        方法一：中心擴展法（推薦）
        時間複雜度: O(n²)
        空間複雜度: O(1)
        """
        if not s:
            return ""
        
        start = 0
        max_len = 0
        
        def expand_around_center(left: int, right: int) -> int:
            """從中心向兩邊擴展，返回回文長度"""
            while left >= 0 and right < len(s) and s[left] == s[right]:
                left -= 1
                right += 1
            return right - left - 1
        
        for i in range(len(s)):
            # 奇數長度回文（中心是一個字符）
            len1 = expand_around_center(i, i)
            # 偶數長度回文（中心是兩個字符）
            len2 = expand_around_center(i, i + 1)
            
            # 取較長的回文
            current_len = max(len1, len2)
            
            # 更新最長回文
            if current_len > max_len:
                max_len = current_len
                # 計算起始位置
                start = i - (current_len - 1) // 2
        
        return s[start:start + max_len]
    
    def longestPalindrome_dp(self, s: str) -> str:
        """
        方法二：動態規劃
        時間複雜度: O(n²)
        空間複雜度: O(n²)
        """
        n = len(s)
        if n < 2:
            return s
        
        # dp[i][j] 表示 s[i:j+1] 是否為回文
        dp = [[False] * n for _ in range(n)]
        
        # 單個字符都是回文
        for i in range(n):
            dp[i][i] = True
        
        start = 0
        max_len = 1
        
        # 按長度遍歷
        for length in range(2, n + 1):
            for i in range(n - length + 1):
                j = i + length - 1
                
                if s[i] == s[j]:
                    if length <= 3:
                        # 長度 2 或 3，只需首尾相同
                        dp[i][j] = True
                    else:
                        # 長度 > 3，還要內部是回文
                        dp[i][j] = dp[i + 1][j - 1]
                
                # 更新最長回文
                if dp[i][j] and length > max_len:
                    start = i
                    max_len = length
        
        return s[start:start + max_len]
#include <string>
#include <vector>

class Solution {
public:
    /**
     * 方法一：中心擴展法（推薦）
     * 時間複雜度: O(n²)
     * 空間複雜度: O(1)
     */
    std::string longestPalindrome(std::string s) {
        if (s.empty()) {
            return "";
        }
        
        int start = 0;
        int maxLen = 0;
        
        for (int i = 0; i < s.length(); i++) {
            // 奇數長度回文
            int len1 = expandAroundCenter(s, i, i);
            // 偶數長度回文
            int len2 = expandAroundCenter(s, i, i + 1);
            
            int currentLen = std::max(len1, len2);
            
            if (currentLen > maxLen) {
                maxLen = currentLen;
                start = i - (currentLen - 1) / 2;
            }
        }
        
        return s.substr(start, maxLen);
    }
    
private:
    int expandAroundCenter(const std::string& s, int left, int right) {
        while (left >= 0 && right < s.length() && s[left] == s[right]) {
            left--;
            right++;
        }
        return right - left - 1;
    }

public:
    /**
     * 方法二：動態規劃
     * 時間複雜度: O(n²)
     * 空間複雜度: O(n²)
     */
    std::string longestPalindrome_dp(std::string s) {
        int n = s.length();
        if (n < 2) {
            return s;
        }
        
        std::vector> dp(n, std::vector(n, false));
        
        // 單個字符都是回文
        for (int i = 0; i < n; i++) {
            dp[i][i] = true;
        }
        
        int start = 0;
        int maxLen = 1;
        
        // 按長度遍歷
        for (int length = 2; length <= n; length++) {
            for (int i = 0; i <= n - length; i++) {
                int j = i + length - 1;
                
                if (s[i] == s[j]) {
                    if (length <= 3) {
                        dp[i][j] = true;
                    } else {
                        dp[i][j] = dp[i + 1][j - 1];
                    }
                }
                
                if (dp[i][j] && length > maxLen) {
                    start = i;
                    maxLen = length;
                }
            }
        }
        
        return s.substr(start, maxLen);
    }
};
class Solution {
    /**
     * 方法一：中心擴展法（推薦）
     * 時間複雜度: O(n²)
     * 空間複雜度: O(1)
     */
    public String longestPalindrome(String s) {
        if (s == null || s.length() == 0) {
            return "";
        }
        
        int start = 0;
        int maxLen = 0;
        
        for (int i = 0; i < s.length(); i++) {
            // 奇數長度回文
            int len1 = expandAroundCenter(s, i, i);
            // 偶數長度回文
            int len2 = expandAroundCenter(s, i, i + 1);
            
            int currentLen = Math.max(len1, len2);
            
            if (currentLen > maxLen) {
                maxLen = currentLen;
                start = i - (currentLen - 1) / 2;
            }
        }
        
        return s.substring(start, start + maxLen);
    }
    
    private int expandAroundCenter(String s, int left, int right) {
        while (left >= 0 && right < s.length() && s.charAt(left) == s.charAt(right)) {
            left--;
            right++;
        }
        return right - left - 1;
    }
    
    /**
     * 方法二：動態規劃
     * 時間複雜度: O(n²)
     * 空間複雜度: O(n²)
     */
    public String longestPalindrome_dp(String s) {
        int n = s.length();
        if (n < 2) {
            return s;
        }
        
        boolean[][] dp = new boolean[n][n];
        
        // 單個字符都是回文
        for (int i = 0; i < n; i++) {
            dp[i][i] = true;
        }
        
        int start = 0;
        int maxLen = 1;
        
        // 按長度遍歷
        for (int length = 2; length <= n; length++) {
            for (int i = 0; i <= n - length; i++) {
                int j = i + length - 1;
                
                if (s.charAt(i) == s.charAt(j)) {
                    if (length <= 3) {
                        dp[i][j] = true;
                    } else {
                        dp[i][j] = dp[i + 1][j - 1];
                    }
                }
                
                if (dp[i][j] && length > maxLen) {
                    start = i;
                    maxLen = length;
                }
            }
        }
        
        return s.substring(start, start + maxLen);
    }
}

複雜度分析

方法一：中心擴展法
  • 時間複雜度：O(n²) - n 個中心點，每個最多擴展 n 次
  • 空間複雜度：O(1) - 只用常數個變數
方法二：動態規劃
  • 時間複雜度：O(n²) - 填充 dp 表
  • 空間複雜度：O(n²) - dp 陣列

重點提示

  • 中心擴展法是面試首選：代碼簡潔、空間 O(1)
  • 兩種回文：別忘記奇數和偶數長度都要檢查
  • 起始位置計算：start = i - (len - 1) // 2
  • 邊界情況：
    • 空字串 → ""
    • 單個字符 → 該字符
    • 所有字符相同 → 整個字串
    • 無回文（長度>1）→ 任意單個字符
  • 優化技巧：
    • 提前剪枝：剩餘長度不足以超過當前最長
    • 從中間開始搜尋（可能更快找到長回文）
  • 面試技巧：
    • 先說中心擴展法（推薦）
    • 提到 DP 解法展現廣度
    • 可以提 Manacher 算法加分
  • 變形題：
    • Palindromic Substrings (#647) - 統計所有回文
    • Valid Palindrome (#125) - 判斷是否回文
    • Longest Palindromic Subsequence (#516) - 子序列而非子字串
