104. Maximum Depth of Binary Tree
Easy

題目描述

給定一個二元樹的根節點 root，返回其最大深度。

二元樹的最大深度是指從根節點到最遠葉子節點的最長路徑上的節點數量。

範例 1:
輸入: root = [3,9,20,null,null,15,7]
輸出: 3

解釋:
     3
   /   \
  9    20
      /  \
     15   7

從根節點 3 到葉子節點 15 或 7 的路徑最長，深度為 3
範例 2:
輸入: root = [1,null,2]
輸出: 2

解釋:
  1
   \
    2

從根節點 1 到葉子節點 2，深度為 2
範例 3:
輸入: root = []
輸出: 0

解釋: 空樹的深度為 0

解題思路

方法一：遞迴解法（DFS - 深度優先搜尋）

使用遞迴的方式計算左右子樹的最大深度，然後取較大值加 1。這是最直觀且簡潔的解法。

核心思想：

樹的深度 = max(左子樹深度, 右子樹深度) + 1

步驟：

  1. 基礎情況：如果節點為空，返回深度 0
  2. 遞迴計算左子樹的最大深度
  3. 遞迴計算右子樹的最大深度
  4. 返回 max(左子樹深度, 右子樹深度) + 1

視覺化範例：[3,9,20,null,null,15,7]

步驟 1: 從根節點 3 開始
     3 (深度=?)
   /   \
  9    20

步驟 2: 計算左子樹深度
  9 是葉子節點
  左子樹深度 = 1

步驟 3: 計算右子樹深度
    20 (深度=?)
   /  \
  15   7
  
  15 是葉子節點，深度 = 1
  7 是葉子節點，深度 = 1
  右子樹深度 = max(1, 1) + 1 = 2

步驟 4: 計算總深度
  總深度 = max(1, 2) + 1 = 3
方法二：迭代解法（BFS - 廣度優先搜尋）

使用層序遍歷（Level Order Traversal）配合佇列，計算樹有多少層。

步驟：

  1. 如果根節點為空，返回 0
  2. 創建一個佇列，將根節點加入佇列
  3. 初始化深度為 0
  4. 當佇列不為空時：
    • 深度 +1
    • 記錄當前層的節點數量
    • 處理當前層的所有節點：
      • 取出節點
      • 將其左右子節點加入佇列
  5. 返回深度
方法三：DFS 迭代（使用堆疊）

使用堆疊模擬遞迴過程，同時記錄每個節點的深度。

步驟：

  1. 使用堆疊存儲 (節點, 深度) 對
  2. 追蹤遇到的最大深度
  3. 處理每個節點時，更新最大深度
  4. 將子節點及其深度加入堆疊

程式碼實現

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def maxDepth(self, root: Optional[TreeNode]) -> int:
        """
        方法一：遞迴解法（DFS）
        時間複雜度: O(n) - 訪問每個節點一次
        空間複雜度: O(h) - 遞迴堆疊深度，h 為樹高
        """
        # 基礎情況：空節點深度為 0
        if not root:
            return 0
        
        # 遞迴計算左右子樹深度
        left_depth = self.maxDepth(root.left)
        right_depth = self.maxDepth(root.right)
        
        # 返回較大深度 + 1（當前節點）
        return max(left_depth, right_depth) + 1
    
    def maxDepth_bfs(self, root: Optional[TreeNode]) -> int:
        """
        方法二：BFS 迭代解法
        時間複雜度: O(n)
        空間複雜度: O(w) - w 為樹的最大寬度
        """
        if not root:
            return 0
        
        from collections import deque
        queue = deque([root])
        depth = 0
        
        while queue:
            depth += 1
            level_size = len(queue)
            
            # 處理當前層的所有節點
            for _ in range(level_size):
                node = queue.popleft()
                
                # 將下一層的節點加入佇列
                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)
        
        return depth
    
    def maxDepth_dfs_iterative(self, root: Optional[TreeNode]) -> int:
        """
        方法三：DFS 迭代解法（使用堆疊）
        時間複雜度: O(n)
        空間複雜度: O(h)
        """
        if not root:
            return 0
        
        stack = [(root, 1)]  # (節點, 深度)
        max_depth = 0
        
        while stack:
            node, depth = stack.pop()
            max_depth = max(max_depth, depth)
            
            # 將子節點及其深度加入堆疊
            if node.left:
                stack.append((node.left, depth + 1))
            if node.right:
                stack.append((node.right, depth + 1))
        
        return max_depth
#include <queue>
#include <stack>
#include <algorithm>

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */

class Solution {
public:
    /**
     * 方法一：遞迴解法（DFS）
     * 時間複雜度: O(n)
     * 空間複雜度: O(h)
     */
    int maxDepth(TreeNode* root) {
        // 基礎情況：空節點深度為 0
        if (root == nullptr) {
            return 0;
        }
        
        // 遞迴計算左右子樹深度
        int leftDepth = maxDepth(root->left);
        int rightDepth = maxDepth(root->right);
        
        // 返回較大深度 + 1
        return std::max(leftDepth, rightDepth) + 1;
    }
    
    /**
     * 方法二：BFS 迭代解法
     * 時間複雜度: O(n)
     * 空間複雜度: O(w)
     */
    int maxDepth_bfs(TreeNode* root) {
        if (root == nullptr) {
            return 0;
        }
        
        std::queue queue;
        queue.push(root);
        int depth = 0;
        
        while (!queue.empty()) {
            depth++;
            int levelSize = queue.size();
            
            // 處理當前層的所有節點
            for (int i = 0; i < levelSize; i++) {
                TreeNode* node = queue.front();
                queue.pop();
                
                // 將下一層的節點加入佇列
                if (node->left != nullptr) {
                    queue.push(node->left);
                }
                if (node->right != nullptr) {
                    queue.push(node->right);
                }
            }
        }
        
        return depth;
    }
    
    /**
     * 方法三：DFS 迭代解法
     * 時間複雜度: O(n)
     * 空間複雜度: O(h)
     */
    int maxDepth_dfs_iterative(TreeNode* root) {
        if (root == nullptr) {
            return 0;
        }
        
        std::stack> stack;
        stack.push({root, 1});
        int maxDepth = 0;
        
        while (!stack.empty()) {
            auto [node, depth] = stack.top();
            stack.pop();
            maxDepth = std::max(maxDepth, depth);
            
            // 將子節點及其深度加入堆疊
            if (node->left != nullptr) {
                stack.push({node->left, depth + 1});
            }
            if (node->right != nullptr) {
                stack.push({node->right, depth + 1});
            }
        }
        
        return maxDepth;
    }
};
import java.util.LinkedList;
import java.util.Queue;
import java.util.Stack;

/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    /**
     * 方法一：遞迴解法（DFS）
     * 時間複雜度: O(n)
     * 空間複雜度: O(h)
     */
    public int maxDepth(TreeNode root) {
        // 基礎情況：空節點深度為 0
        if (root == null) {
            return 0;
        }
        
        // 遞迴計算左右子樹深度
        int leftDepth = maxDepth(root.left);
        int rightDepth = maxDepth(root.right);
        
        // 返回較大深度 + 1
        return Math.max(leftDepth, rightDepth) + 1;
    }
    
    /**
     * 方法二：BFS 迭代解法
     * 時間複雜度: O(n)
     * 空間複雜度: O(w)
     */
    public int maxDepth_bfs(TreeNode root) {
        if (root == null) {
            return 0;
        }
        
        Queue queue = new LinkedList<>();
        queue.offer(root);
        int depth = 0;
        
        while (!queue.isEmpty()) {
            depth++;
            int levelSize = queue.size();
            
            // 處理當前層的所有節點
            for (int i = 0; i < levelSize; i++) {
                TreeNode node = queue.poll();
                
                // 將下一層的節點加入佇列
                if (node.left != null) {
                    queue.offer(node.left);
                }
                if (node.right != null) {
                    queue.offer(node.right);
                }
            }
        }
        
        return depth;
    }
    
    /**
     * 方法三：DFS 迭代解法
     * 時間複雜度: O(n)
     * 空間複雜度: O(h)
     */
    public int maxDepth_dfs_iterative(TreeNode root) {
        if (root == null) {
            return 0;
        }
        
        Stack stack = new Stack<>();
        stack.push(new Pair(root, 1));
        int maxDepth = 0;
        
        while (!stack.isEmpty()) {
            Pair pair = stack.pop();
            TreeNode node = pair.node;
            int depth = pair.depth;
            maxDepth = Math.max(maxDepth, depth);
            
            // 將子節點及其深度加入堆疊
            if (node.left != null) {
                stack.push(new Pair(node.left, depth + 1));
            }
            if (node.right != null) {
                stack.push(new Pair(node.right, depth + 1));
            }
        }
        
        return maxDepth;
    }
    
    // 輔助類別用於存儲節點和深度
    private class Pair {
        TreeNode node;
        int depth;
        
        Pair(TreeNode node, int depth) {
            this.node = node;
            this.depth = depth;
        }
    }
}

複雜度分析

時間複雜度: O(n)

所有方法都需要訪問每個節點恰好一次，其中 n 是二元樹的節點數量。

空間複雜度:
  • 遞迴 DFS：O(h)，其中 h 是樹的高度。遞迴堆疊的深度等於樹的高度
  • BFS：O(w)，其中 w 是樹的最大寬度。佇列中最多存儲一層的所有節點
  • 迭代 DFS：O(h)，堆疊中最多存儲從根到葉的一條路徑

註：

  • 對於完全不平衡的樹（鏈狀）：h = n
  • 對於完全平衡的二元樹：h = log(n)
  • 對於完全二元樹：w ≈ n/2（最底層）

重點提示

  • 遞迴基礎：這是學習樹遞迴的最基礎題目，務必掌握
  • 深度定義：深度從 1 開始計數（包含根節點），空樹深度為 0
  • 高度 vs 深度：在這題中兩者相同，但定義上深度從根往下算，高度從葉往上算
  • 方法選擇：
    • 遞迴 DFS：最簡潔，推薦用於面試
    • BFS：適合需要層序處理的場景
    • 迭代 DFS：適合避免遞迴的場景
  • 擴展問題：可以輕鬆修改為求最小深度（Minimum Depth）
