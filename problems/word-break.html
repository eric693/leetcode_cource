139. Word Break
Medium

題目描述

給定一個字串 s 和一個字串字典 wordDict，如果 s 可以被空格分割成一個或多個字典中出現的單詞，則返回 true。

注意：字典中的同一個單詞可能在分割中被重複使用多次。

範例 1:
輸入: s = "leetcode", wordDict = ["leet","code"]
輸出: true
解釋: "leetcode" 可以分割為 "leet code"。
範例 2:
輸入: s = "applepenapple", wordDict = ["apple","pen"]
輸出: true
解釋: "applepenapple" 可以分割為 "apple pen apple"。
     注意你可以重複使用字典中的單詞。
範例 3:
輸入: s = "catsandog", wordDict = ["cats","dog","sand","and","cat"]
輸出: false
限制條件:
1 <= s.length <= 300
1 <= wordDict.length <= 1000
1 <= wordDict[i].length <= 20
s 和 wordDict[i] 僅由小寫英文字母組成
wordDict 中的所有字串互不相同

解題思路

方法一：動態規劃（推薦）⭐⭐

使用動態規劃，dp[i] 表示字串前 i 個字符能否被字典中的單詞分割。

核心思想：

如果 dp[j] 為 true（前 j 個字符可以分割）且 s[j:i] 在字典中，則 dp[i] 為 true。

狀態定義：

  • dp[i]：字串 s 的前 i 個字符能否被分割
  • 初始狀態：dp[0] = true（空字串可以被分割）
  • 轉移方程：dp[i] = dp[j] && (s[j:i] in wordDict)，其中 0 ≤ j < i

步驟：

  1. 創建 dp 陣列，長度為 n+1，初始化 dp[0] = true
  2. 將 wordDict 轉為 Set 加速查找
  3. 雙重循環：
    • 外層：i 從 1 到 n
    • 內層：j 從 0 到 i-1
    • 如果 dp[j] && s[j:i] in wordSet，則 dp[i] = true
  4. 返回 dp[n]

方法二：記憶化搜尋（DFS + Memo）

使用 DFS 遞迴搜尋，配合記憶化避免重複計算。

思路：

  • 從索引 start 開始，嘗試所有可能的單詞
  • 如果 s[start:end] 在字典中，遞迴檢查剩餘部分
  • 使用 memo 記錄已計算的結果
優化技巧
  • 使用 Set：將 wordDict 轉為 Set，查找時間 O(1)
  • 提前終止：找到一個可行分割即可 break
  • 最大長度優化：記錄字典中最長單詞，限制內層循環範圍
  • BFS 方法：也可以使用 BFS + Queue 實現

程式碼實現

class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        """
        方法一：動態規劃（推薦）
        時間複雜度: O(n^2 * m)，n 是字串長度，m 是平均單詞長度
        空間複雜度: O(n)
        """
        n = len(s)
        # 轉為 Set 加速查找
        word_set = set(wordDict)
        
        # dp[i] 表示 s[0:i] 能否被分割
        dp = [False] * (n + 1)
        dp[0] = True  # 空字串
        
        # 遍歷所有位置
        for i in range(1, n + 1):
            # 嘗試所有可能的分割點
            for j in range(i):
                # 如果前 j 個字符可以分割，且 s[j:i] 在字典中
                if dp[j] and s[j:i] in word_set:
                    dp[i] = True
                    break  # 找到一個即可
        
        return dp[n]
    
    def wordBreak_optimized(self, s: str, wordDict: List[str]) -> bool:
        """
        優化版：限制內層循環範圍
        時間複雜度: O(n * max_len * m)
        """
        n = len(s)
        word_set = set(wordDict)
        max_len = max(len(word) for word in wordDict)
        
        dp = [False] * (n + 1)
        dp[0] = True
        
        for i in range(1, n + 1):
            # 只檢查最大單詞長度範圍內的分割點
            for j in range(max(0, i - max_len), i):
                if dp[j] and s[j:i] in word_set:
                    dp[i] = True
                    break
        
        return dp[n]
    
    def wordBreak_dfs(self, s: str, wordDict: List[str]) -> bool:
        """
        方法二：DFS + 記憶化
        時間複雜度: O(n^2 * m)
        空間複雜度: O(n)
        """
        word_set = set(wordDict)
        memo = {}
        
        def dfs(start):
            # 已到達字串末尾
            if start == len(s):
                return True
            
            # 已計算過
            if start in memo:
                return memo[start]
            
            # 嘗試所有可能的單詞
            for end in range(start + 1, len(s) + 1):
                if s[start:end] in word_set:
                    if dfs(end):
                        memo[start] = True
                        return True
            
            memo[start] = False
            return False
        
        return dfs(0)
    
    def wordBreak_bfs(self, s: str, wordDict: List[str]) -> bool:
        """
        方法三：BFS
        時間複雜度: O(n^2 * m)
        空間複雜度: O(n)
        """
        from collections import deque
        
        word_set = set(wordDict)
        queue = deque([0])
        visited = {0}
        
        while queue:
            start = queue.popleft()
            
            # 已到達末尾
            if start == len(s):
                return True
            
            # 嘗試所有可能的結束位置
            for end in range(start + 1, len(s) + 1):
                if end in visited:
                    continue
                    
                if s[start:end] in word_set:
                    if end == len(s):
                        return True
                    queue.append(end)
                    visited.add(end)
        
        return False
#include <vector>
#include <string>
#include <unordered_set>
using namespace std;

class Solution {
public:
    /**
     * 方法一：動態規劃
     * 時間複雜度: O(n^2 * m)
     * 空間複雜度: O(n)
     */
    bool wordBreak(string s, vector& wordDict) {
        int n = s.length();
        unordered_set wordSet(wordDict.begin(), wordDict.end());
        
        vector dp(n + 1, false);
        dp[0] = true;
        
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < i; j++) {
                if (dp[j] && wordSet.count(s.substr(j, i - j))) {
                    dp[i] = true;
                    break;
                }
            }
        }
        
        return dp[n];
    }
    
    /**
     * 優化版：限制內層循環範圍
     */
    bool wordBreak_optimized(string s, vector& wordDict) {
        int n = s.length();
        unordered_set wordSet(wordDict.begin(), wordDict.end());
        
        // 找出最長單詞長度
        int maxLen = 0;
        for (const string& word : wordDict) {
            maxLen = max(maxLen, (int)word.length());
        }
        
        vector dp(n + 1, false);
        dp[0] = true;
        
        for (int i = 1; i <= n; i++) {
            for (int j = max(0, i - maxLen); j < i; j++) {
                if (dp[j] && wordSet.count(s.substr(j, i - j))) {
                    dp[i] = true;
                    break;
                }
            }
        }
        
        return dp[n];
    }
};
import java.util.*;

class Solution {
    /**
     * 方法一：動態規劃
     * 時間複雜度: O(n^2 * m)
     * 空間複雜度: O(n)
     */
    public boolean wordBreak(String s, List wordDict) {
        int n = s.length();
        Set wordSet = new HashSet<>(wordDict);
        
        boolean[] dp = new boolean[n + 1];
        dp[0] = true;
        
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < i; j++) {
                if (dp[j] && wordSet.contains(s.substring(j, i))) {
                    dp[i] = true;
                    break;
                }
            }
        }
        
        return dp[n];
    }
    
    /**
     * 優化版：限制內層循環範圍
     */
    public boolean wordBreak_optimized(String s, List wordDict) {
        int n = s.length();
        Set wordSet = new HashSet<>(wordDict);
        
        // 找出最長單詞長度
        int maxLen = 0;
        for (String word : wordDict) {
            maxLen = Math.max(maxLen, word.length());
        }
        
        boolean[] dp = new boolean[n + 1];
        dp[0] = true;
        
        for (int i = 1; i <= n; i++) {
            for (int j = Math.max(0, i - maxLen); j < i; j++) {
                if (dp[j] && wordSet.contains(s.substring(j, i))) {
                    dp[i] = true;
                    break;
                }
            }
        }
        
        return dp[n];
    }
}

複雜度分析

時間複雜度: O(n² × m)

n 是字串長度，m 是平均單詞長度。雙重循環 O(n²)，substring 比較 O(m)。 優化後可以降到 O(n × max_len × m)。

空間複雜度: O(n)

dp 陣列需要 O(n) 空間，wordSet 需要 O(k × m) 空間（k 是字典大小）。

重點提示

  • 動態規劃思路：從前往後，檢查每個位置是否可以被分割
  • 關鍵優化：
    • 將 wordDict 轉為 Set，查找 O(1)
    • 使用 break 提前終止內層循環
    • 記錄最大單詞長度，限制搜尋範圍
  • 三種解法比較：
    • DP：最容易理解，效率高
    • DFS + Memo：代碼簡潔，但可能棧溢出
    • BFS：適合找最短分割
  • 邊界情況：
    • 空字串 → true
    • 字典為空 → false（除非 s 為空）
    • 單詞可以重複使用
    • s 比所有單詞都長
  • 常見錯誤：
    • 忘記 dp[0] = true
    • substring 邊界處理錯誤
    • 沒有使用 Set 導致超時
  • 面試技巧：
    • 先說 DP 思路：dp[i] 的定義
    • 畫圖展示狀態轉移
    • 提到優化：Set、break、最大長度
    • 可以討論其他解法（DFS、BFS）
