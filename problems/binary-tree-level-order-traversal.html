102. Binary Tree Level Order Traversal
Medium

題目描述

給定一個二元樹的根節點 root，返回其節點值的層序遍歷（從左到右，逐層遍歷）。

範例 1:
輸入: root = [3,9,20,null,null,15,7]
輸出: [[3],[9,20],[15,7]]

視覺化:
       3
      / \
     9  20
       /  \
      15   7

層序遍歷:
第 0 層: [3]
第 1 層: [9, 20]
第 2 層: [15, 7]
範例 2:
輸入: root = [1]
輸出: [[1]]
範例 3:
輸入: root = []
輸出: []
限制條件:
樹中節點數目在範圍 [0, 2000] 內
-1000 <= Node.val <= 1000

解題思路

方法一：BFS（廣度優先搜尋）推薦 ⭐

使用佇列（Queue）進行層序遍歷，這是最自然的解法。

核心思想：

使用佇列維護每一層的節點，處理完一層後再處理下一層。

步驟：

  1. 將根節點加入佇列
  2. 當佇列不為空時：
    • 記錄當前層的節點數量 size
    • 創建當前層的結果陣列
    • 處理 size 個節點：
      • 取出節點並加入結果陣列
      • 將左右子節點加入佇列
    • 將當前層結果加入最終結果

詳細執行過程：

       3
      / \
     9  20
       /  \
      15   7

初始: queue = [3], result = []

第 1 輪（處理第 0 層）:
  size = 1
  取出 3 → level = [3]
  加入子節點: queue = [9, 20]
  result = [[3]]

第 2 輪（處理第 1 層）:
  size = 2
  取出 9 → level = [9]
  取出 20 → level = [9, 20]
  加入子節點: queue = [15, 7]
  result = [[3], [9, 20]]

第 3 輪（處理第 2 層）:
  size = 2
  取出 15 → level = [15]
  取出 7 → level = [15, 7]
  queue = []
  result = [[3], [9, 20], [15, 7]]

完成！
方法二：DFS（深度優先搜尋）

使用遞迴記錄每個節點所在的層級。

核心思想：

傳遞當前深度參數，將節點值加入對應層級的陣列。

步驟：

  1. 創建結果陣列
  2. 遞迴函數接收節點和深度：
    • 如果深度等於結果陣列長度，創建新層陣列
    • 將當前節點值加入對應層
    • 遞迴處理左右子樹（深度 +1）

DFS 訪問順序：

       3 (depth=0)
      / \
     9   20 (depth=1)
        /  \
       15   7 (depth=2)

訪問順序: 3 → 9 → 20 → 15 → 7

result[0] = [3]       (depth=0 訪問 3)
result[1] = [9, 20]   (depth=1 訪問 9, 20)
result[2] = [15, 7]   (depth=2 訪問 15, 7)
BFS vs DFS 對比
特性 BFS（推薦） DFS
實現方式 佇列（迭代） 遞迴
直觀程度 ⭐⭐⭐⭐⭐ 很直觀 ⭐⭐⭐ 需要思考
代碼長度 稍長 簡潔
空間複雜度 O(n) 佇列 O(h) 遞迴堆疊

程式碼實現

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

from collections import deque
from typing import List, Optional

class Solution:
    def levelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        """
        方法一：BFS（推薦）
        時間複雜度: O(n)
        空間複雜度: O(n)
        """
        if not root:
            return []
        
        result = []
        queue = deque([root])
        
        while queue:
            # 當前層的節點數量
            level_size = len(queue)
            current_level = []
            
            # 處理當前層的所有節點
            for _ in range(level_size):
                node = queue.popleft()
                current_level.append(node.val)
                
                # 將下一層的節點加入佇列
                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)
            
            result.append(current_level)
        
        return result
    
    def levelOrder_dfs(self, root: Optional[TreeNode]) -> List[List[int]]:
        """
        方法二：DFS
        時間複雜度: O(n)
        空間複雜度: O(h) - h 為樹高
        """
        result = []
        
        def dfs(node, depth):
            if not node:
                return
            
            # 如果當前深度等於結果陣列長度，創建新層
            if depth == len(result):
                result.append([])
            
            # 將當前節點值加入對應層
            result[depth].append(node.val)
            
            # 遞迴處理左右子樹
            dfs(node.left, depth + 1)
            dfs(node.right, depth + 1)
        
        dfs(root, 0)
        return result
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */

#include <vector>
#include <queue>

class Solution {
public:
    /**
     * 方法一：BFS（推薦）
     * 時間複雜度: O(n)
     * 空間複雜度: O(n)
     */
    std::vector> levelOrder(TreeNode* root) {
        if (!root) {
            return {};
        }
        
        std::vector> result;
        std::queue queue;
        queue.push(root);
        
        while (!queue.empty()) {
            // 當前層的節點數量
            int levelSize = queue.size();
            std::vector currentLevel;
            
            // 處理當前層的所有節點
            for (int i = 0; i < levelSize; i++) {
                TreeNode* node = queue.front();
                queue.pop();
                currentLevel.push_back(node->val);
                
                // 將下一層的節點加入佇列
                if (node->left) {
                    queue.push(node->left);
                }
                if (node->right) {
                    queue.push(node->right);
                }
            }
            
            result.push_back(currentLevel);
        }
        
        return result;
    }
    
    /**
     * 方法二：DFS
     * 時間複雜度: O(n)
     * 空間複雜度: O(h)
     */
    std::vector> levelOrder_dfs(TreeNode* root) {
        std::vector> result;
        dfs(root, 0, result);
        return result;
    }
    
private:
    void dfs(TreeNode* node, int depth, std::vector>& result) {
        if (!node) {
            return;
        }
        
        // 如果當前深度等於結果陣列長度，創建新層
        if (depth == result.size()) {
            result.push_back({});
        }
        
        // 將當前節點值加入對應層
        result[depth].push_back(node->val);
        
        // 遞迴處理左右子樹
        dfs(node->left, depth + 1, result);
        dfs(node->right, depth + 1, result);
    }
};
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

import java.util.*;

class Solution {
    /**
     * 方法一：BFS（推薦）
     * 時間複雜度: O(n)
     * 空間複雜度: O(n)
     */
    public List> levelOrder(TreeNode root) {
        List> result = new ArrayList<>();
        
        if (root == null) {
            return result;
        }
        
        Queue queue = new LinkedList<>();
        queue.offer(root);
        
        while (!queue.isEmpty()) {
            // 當前層的節點數量
            int levelSize = queue.size();
            List currentLevel = new ArrayList<>();
            
            // 處理當前層的所有節點
            for (int i = 0; i < levelSize; i++) {
                TreeNode node = queue.poll();
                currentLevel.add(node.val);
                
                // 將下一層的節點加入佇列
                if (node.left != null) {
                    queue.offer(node.left);
                }
                if (node.right != null) {
                    queue.offer(node.right);
                }
            }
            
            result.add(currentLevel);
        }
        
        return result;
    }
    
    /**
     * 方法二：DFS
     * 時間複雜度: O(n)
     * 空間複雜度: O(h)
     */
    public List> levelOrder_dfs(TreeNode root) {
        List> result = new ArrayList<>();
        dfs(root, 0, result);
        return result;
    }
    
    private void dfs(TreeNode node, int depth, List> result) {
        if (node == null) {
            return;
        }
        
        // 如果當前深度等於結果陣列長度，創建新層
        if (depth == result.size()) {
            result.add(new ArrayList<>());
        }
        
        // 將當前節點值加入對應層
        result.get(depth).add(node.val);
        
        // 遞迴處理左右子樹
        dfs(node.left, depth + 1, result);
        dfs(node.right, depth + 1, result);
    }
}

複雜度分析

時間複雜度: O(n)

需要訪問樹中的每個節點一次，其中 n 是節點總數。

空間複雜度:
  • BFS：O(n) - 佇列最多存儲一層的所有節點
  • DFS：O(h) - 遞迴堆疊深度等於樹高 h

重點提示

  • BFS 是層序遍歷的標準解法：最直觀、最容易理解
  • 關鍵技巧：記錄當前層的節點數量 size
  • Python 提示：使用 collections.deque，比 list 快
  • 邊界情況：
    • 空樹 → 返回 []
    • 只有根節點 → 返回 [[root.val]]
    • 完全二元樹 vs 斜樹
  • 常見錯誤：
    • 忘記記錄當前層大小
    • 提前檢查子節點是否為 null
    • DFS 中忘記創建新層陣列
  • 面試技巧：
    • 先畫圖展示層序遍歷過程
    • 說明 BFS 用佇列實現
    • 提到可以用 DFS 但 BFS 更直觀
